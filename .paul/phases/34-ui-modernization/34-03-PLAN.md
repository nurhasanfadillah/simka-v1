    Buat badge.tsx dengan CVA variants:
    ```
    import { cva, type VariantProps } from "class-variance-authority"
    import { cn } from "@/lib/utils"

    const badgeVariants = cva(
      "inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium transition-colors",
      {
        variants: {
          variant: {
            default: "bg-primary text-primary-foreground",
            secondary: "bg-secondary text-secondary-foreground",
            success: "bg-emerald-100 text-emerald-800",
            warning: "bg-amber-100 text-amber-800",
            danger: "bg-red-100 text-red-800",
            outline: "border text-foreground",
          },
        },
        defaultVariants: { variant: "default" },
      }
    )

    interface BadgeProps
      extends React.HTMLAttributes<HTMLDivElement>,
        VariantProps<typeof badgeVariants> {}

    function Badge({ className, variant, ...props }: BadgeProps) {
      return <div className={cn(badgeVariants({ variant }), className)} {...props} />
    }

    export { Badge, badgeVariants }
    ```
    Definisikan variant warna langsung (tidak via CSS var) karena simple.
    Radius: rounded-full (pill shape) — sesuai user request: round hanya di card & input, tapi badge pill adalah exception untuk status indicator.
  </action>
  <verify>pnpm --filter frontend build</verify>
