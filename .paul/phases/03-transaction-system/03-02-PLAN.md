    **3. Buat `apps/backend/src/transactions/transactions.controller.ts`:**
    ```typescript
    import {
      Controller, Get, Post, Param, Query, Body, ParseIntPipe, Request,
    } from '@nestjs/common';
    import { TransactionsService } from './transactions.service';
    import { CreateTransactionDto } from './dto/create-transaction.dto';
    import { RequirePermissions } from '../auth/decorators/require-permissions.decorator';

    @Controller('transactions')
    export class TransactionsController {
      constructor(private readonly service: TransactionsService) {}

      @Post()
      @RequirePermissions('transaction.create')
      create(@Body() dto: CreateTransactionDto, @Request() req: any) {
        return this.service.create(dto, req.user.sub);
      }

      @Get()
      @RequirePermissions('transaction.view')
      findAll(
        @Query('studentId') studentId?: string,
        @Query('status') status?: 'aktif' | 'void',
        @Query('dateFrom') dateFrom?: string,
        @Query('dateTo') dateTo?: string,
      ) {
        return this.service.findAll({
          studentId: studentId ? parseInt(studentId, 10) : undefined,
          status,
          dateFrom,
          dateTo,
        });
      }

      @Get(':id')
      @RequirePermissions('transaction.view')
      findOne(@Param('id', ParseIntPipe) id: number) {
        return this.service.findOne(id);
      }
    }
    ```

<task type="auto">
  <name>Task 2: TransactionsService — create() dengan auto-status update</name>
  <files>apps/backend/src/transactions/transactions.service.ts</files>
  <action>
    Buat `apps/backend/src/transactions/transactions.service.ts` dengan method `create()`:

    **Imports yang dibutuhkan:**
    - NestJS: `Inject, Injectable, BadRequestException`
    - Drizzle: `eq, like, and, SQL, gte, lte, inArray`
    - DB: `DRIZZLE`, `DB`
    - Schema: `transactions, transactionItems, bills, billMonths` dari financial.schema

    **Helper `generateTrxNumber(tx):`**
    ```
    Format: TRX-YYYYMMDD-NNNNN
    Prefix: TRX-{YYYYMMDD} dari tanggal hari ini
    Seq: MAX dari transactions.transactionNumber LIKE 'TRX-YYYYMMDD-%'
         parse 5 digit terakhir, +1, padStart(5, '0')
    Jika tidak ada: '00001'
    ```

    **Method `create(dto, createdById):`**
    Semua dalam `this.db.transaction(async (tx) => { ... })`:
    1. Generate transactionNumber via helper
    2. Hitung totalAmount = sum(items.amount)
    3. Insert ke `transactions` → dapat `transaction.id`
    4. Insert semua items ke `transactionItems` dalam satu `tx.insert().values([...])`
    5. Kumpulkan unique billMonthIds (filter yang tidak null/undefined)
    6. Update bill_months: `status='lunas'`, `paidAt=new Date()` untuk setiap billMonthId
    7. Kumpulkan unique billIds dari items
    8. Untuk setiap billId: hitung ulang bills.status
       - Query semua bill_months milik billId tersebut
       - Jika semua 'lunas' → bills.status = 'lunas'
       - Jika sebagian 'lunas' → bills.status = 'cicilan'
       - Jika tidak ada yang 'lunas' → bills.status = 'belum_bayar'
       - Update bills dengan `updatedAt = new Date()`
    9. Return: `{ id, transactionNumber, totalAmount, status: 'aktif', createdAt }`

    PENTING: Gunakan `tx` (transaction handle) untuk semua query di dalam blok transaction, bukan `this.db`.
    AVOID: Jangan update bill_months yang billMonthId-nya null/undefined (tagihan bebas tidak perlu).
    AVOID: Jangan hapus transaksi — constraint dari PROJECT.md.

  <name>Task 3: TransactionsService — findAll() + findOne() + integration test via curl</name>
  <files>apps/backend/src/transactions/transactions.service.ts</files>
  <action>
    Tambah method `findAll(filters)` dan `findOne(id)` ke TransactionsService:

    **`findAll(filters: { studentId?, status?, dateFrom?, dateTo? })`:**
    - Query dari `transactions` JOIN `students` (via transactions.studentId)
    - Select: id, transactionNumber, studentId, studentName (students.name), nis (students.nis),
              totalAmount, status, notes, createdBy, createdAt
    - Filter conditions (SQL[]):
      - studentId → `eq(transactions.studentId, filters.studentId)`
      - status → `eq(transactions.status, filters.status)`
      - dateFrom → `gte(transactions.createdAt, new Date(filters.dateFrom))`
      - dateTo → `lte(transactions.createdAt, new Date(filters.dateTo + 'T23:59:59'))`
    - Order: `desc(transactions.createdAt)`

    **`findOne(id: number)`:**
    - Query transactions JOIN students WHERE transactions.id = id
    - Jika tidak ditemukan: throw NotFoundException('Transaksi tidak ditemukan')
    - Query transactionItems WHERE transactionId = id
    - Return: `{ ...transaction, items }`

    Import yang perlu ditambah: `desc, gte, lte` dari drizzle-orm, `students` dari academic.schema, `NotFoundException` dari @nestjs/common.
